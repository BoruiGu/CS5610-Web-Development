Week3 - Array of Objects

In this experiment an array of objects is used to fill in contents of a table.

This is how it looks like before contents are filled in.

# a b c

Create a table row for each object in the array. Append table rows to table body.

# a b c 
var content = [
    { c1: "I", c2: "1", c3: "2", c4: "3" },
    { c1: "II", c2: "4", c3: "5", c4: "6" },
    { c1: "III", c2: "7", c3: "8", c4: "9" },
];
var tbody = $("#table tbody");
for (var i in content) {
    var r = content[i];
    var tr = $("<tr>");
    var td = $("<td>");
    td.append(r.c1);
    td.append("</td>");
    tr.append(td);
    td = $("<td>");
    td.append(r.c2);
    td.append("</td>");
    tr.append(td);
    td = $("<td>");
    td.append(r.c3);
    td.append("</td>");
    tr.append(td);
    td = $("<td>");
    td.append(r.c4);
    td.append("</td>");
    tr.append(td);
    tbody.append(tr);
}
